\begin{Exercise}[
name={},

difficulty=0,
origin={\cite{YL}}]
Solve for $X$ given that it satisfies
\[
\Big(\text{tr}(A)A+\trans{X}A\Big)^{-1}=\frac{1}{\det(A)}I_3
\]
where
\[
B=
\begin{mat}
1&2&3\\
0&2&3\\
0&0&3\\
\end{mat}\;
\]

\end{Exercise}

\begin{Answer}
$
X=
\begin{mat}
0 & 0 & 0\\
-6 & -3 & 0\\
0 & -3 & -4
\end{mat}
$
\end{Answer}
