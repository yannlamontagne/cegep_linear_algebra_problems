Given the matrix
\[
	A = \begin{mat} 2 & -1\\ 0 & 2 \end{mat}
\]
%\begin{multicols}{2}
	\Question $A^2$
	\Question $A^3$
	\Question $A^5$ 
	\EndCurrentQuestion
%\end{multicols}
\end{Exercise}

\begin{Answer}
	\Question $A^2 =  \begin{mat} 4 & -4 \\ 0 & 4 \end{mat}$
	\Question $A^3 = A^2 \, A   =  \begin{mat} 8 & -12 \\ 0 & 8 \end{mat}$
	\Question $A^5 = A^3 \, A^2 =  \begin{mat} 32 & -80 \\ 0 & 32 \end{mat}$
\end{Answer}
