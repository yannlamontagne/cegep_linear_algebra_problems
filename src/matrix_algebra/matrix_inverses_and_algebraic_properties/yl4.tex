\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Solve of $A$ given that it satisfies
\[
(I-A^T)^{-1} = (\trace(B)B^2)^T
\]
where
\[
B=
\begin{bmatrix}
2 & 3\\
1 & 2
\end{bmatrix}\;
\]

\end{Exercise}

\begin{Answer}
$
A=
\begin{mat}
-\frac34 & 3\\
1 & -\frac34
\end{mat}
$
\end{Answer}
