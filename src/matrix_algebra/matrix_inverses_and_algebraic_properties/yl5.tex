\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Solve of $X$ given that it satisfies
\[
DXD^T = \trace(BC)BC
\]
where
\[
B=
\begin{mat}
2 & 1 & 0\\
-3 & 4 & 0
\end{mat}\;
C=
\begin{mat}
2 & -1\\
3 & -2\\
1 & 0
\end{mat}
D=
\begin{mat}
2 & -2\\
1 & -2
\end{mat}.
\]

\end{Exercise}

\begin{Answer}
$
A=
\begin{mat}
0 & -1\\
-11 & -\frac{17}{2}
\end{mat}
$
\end{Answer}
