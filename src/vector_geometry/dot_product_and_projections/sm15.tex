\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{SM}}]
Determine all values of $k$ for which the vectors are orthogonal.
\Question $(3,-1)$, $(2,k)$
\Question $(3,-1)$, $(k,k^2)$
\Question $(1,2,3)$, $(3,-k,k)$
\Question $(1,2,3)$, $(k,k,-k)$
\end{Exercise}

\begin{Answer}
\Question $k=6$
\Question $k=0$ or $k=3$
\Question $k=-3$
\Question any $k\in\Re$
\end{Answer}
