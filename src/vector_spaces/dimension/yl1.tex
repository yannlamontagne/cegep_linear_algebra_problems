\begin{Exercise}[
name={},
title={}, 
difficulty=0,
origin={\cite{YL}}]
Given \[W=\set{p(x)=a_0+a_1x+a_2x^2+a_3x^3 \suchthat p(1)=0 \text{ and } p(-1)=0}\] a subspace of $P_3$. Determine the dimension of $W$.

\end{Exercise}

\begin{Answer}
$\set{-1+x^2,\;-x+x^3}$ is a basis of $W$, therefore $W$ is of dimension 2.
\end{Answer}
